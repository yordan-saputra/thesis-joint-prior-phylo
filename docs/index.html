Thesis Draft 1

Joint prior distributions for hierarchical phylogenetic models

Author

Yordan Saputra

Problem Description

Phylogenetic trees are used to represent the evolutionary history between a set of species. The implied dependencies between species can be expressed statistically via phylogenetic models. However, these models are often hard to estimate and may require strong prior knowledge, which we express in Bayesian statistics via prior distributions. The goal of this thesis is to implement and evaluate joint prior distributions for the variance parameters of phylogenetic models. We have concrete ideas for such priors, which we will discuss with you upon starting your thesis.

Project Structure

  • Literature search on priors for phylogenetic models
  • Implement the new and existing priors in Stan
  • Run simulation studies
  • Analyse and discuss results

load_or_fit_brms <- function (data, model_name, formula, family, data2, prior, replace = FALSE) {
  modelpath <- file.path(getwd(), 'PhyloModel', paste0(model_name, '.rds'))
  
  if (file.exists(modelpath) && !replace) {
    model <- readRDS(modelpath)
  } else {
    file.remove(modelpath)
    model <- brm(
      formula, data = data, family = family, 
      data2 = data2, prior = prior, 
      refresh = 0, silent = 2, 
      control = list(adapt_delta = 0.95),
      seed = SEED
    )
    saveRDS(model, modelpath)
  }
  
  return(model)
}

load_or_fit_stan <- function (data, model_name, stan_model, replace = FALSE) {
  modelpath <- file.path(getwd(), 'PhyloModel', paste0(model_name, '.rds'))
  stanpath <- file.path(getwd(), 'PhyloModel', paste0(model_name, '.stan'))
  exepath <- file.path(getwd(), 'PhyloModel', paste0(model_name, '.exe'))
  
  if (file.exists(modelpath) && !replace) {
    model <- readRDS(modelpath)
  } else {
    file.remove(c(modelpath, stanpath, exepath))
    writeLines(stan_model@model_code[1], stanpath)
    stan_model <- cmdstan_model(stanpath)
    
    model <- stan_model$sample(
      data = list(
        N = nrow(data), y = data$phen,
        x1 = data$spec_mean_cf, x2 = data$within_spec_cf,
        J = length(unique(data$species)),
        group = as.numeric(factor(data$species)),
        A_phylo = A
      ),
      chains = 4, parallel_chains = 4,
      iter_warmup = 1000, iter_sampling = 1000,
      adapt_delta = 0.95, max_treedepth = 12,
      seed = SEED,
    )
    model$save_object(file = modelpath)
  }
  
  return(model)
}

parameters {
  // Fixed effects
  real mu; // Population intercept
  real beta_b; // Slope for spec_mean_cf
  real beta_w; // Slope for within_spec_cf
  
  // Variance components
  real<lower=0> sigma_phy; // SD for phylogenetic effect
  real<lower=0> sigma_sp; // SD of species random effects
  
  // Non-centered parameterization
  vector[J] phy_eff_z;// latent variable for phylogenetic
  vector[J] sp_eff_z; // latent variable for species
  
  // Residual error
  real<lower=0> sigma;
}

transformed parameters {
  vector[J] phy_eff = sigma_phy * (cholesky_decompose(A_phylo) * phy_eff_z); // phylogenetic effect (a)
  vector[J] sp_eff = sigma_sp * sp_eff_z; // species effect (s)
}

model {
  // Priors
  mu ~ normal(0, 50);
  beta_b ~ normal(0, 10);
  beta_w ~ normal(0, 10);
  
  sigma_phy ~ student_t(3, 0, 20);
  sigma_sp ~ student_t(3, 0, 20);
  
  sigma ~ student_t(3, 0, 20);
  
  // Random effects priors
  phy_eff_z ~ normal(0, 1);
  sp_eff_z ~ normal(0, 1);

  for (i in 1:N) {
    y[i] ~ normal(mu + beta_b * x1[i] + beta_w * x2[i] + sp_eff[group[i]] + phy_eff[group[i]], sigma);
  }
}

  vector[N] x2; // within_spec_cf variable
  
  // Group-level data
  int<lower=0> J; // Number of groups (species)
  array[N] int<lower=1, upper=J> group;  // Group (species) ID
  
  // Phylogenetic data
  matrix[J, J] A_phylo; // phylogenetic vcv matrix
}

parameters {
  // Fixed effects
  real mu; // Population intercept
  real beta_b; // Slope for spec_mean_cf
  real beta_w; // Slope for within_spec_cf
  
  // Variance components
  real<lower=0> sigma_t; // SD of combined components
  real<lower=0, upper=1> rho; // proportion effect variance
  
  // Non-centered parameterization
  vector[J] phy_eff_z;// latent variable for phylogenetic
  vector[J] sp_eff_z; // latent variable for species
  
  // Residual error
  real<lower=0> sigma;
}

transformed parameters {
  vector[J] phy_eff = sigma_t * sqrt(1 - rho) * (cholesky_decompose(A_phylo) * phy_eff_z);
  vector[J] sp_eff = sigma_t * sqrt(rho) * sp_eff_z;
}

model {
  // Priors
  mu ~ normal(0, 50);
  beta_b ~ normal(0, 10);
  beta_w ~ normal(0, 10);
  
  rho ~ beta(0.5, 0.5);
  sigma_t ~ student_t(3, 0, 20);
  
  sigma ~ student_t(3, 0, 20);
  
  // Random effects priors
  phy_eff_z ~ normal(0, 1);
  sp_eff_z ~ normal(0, 1);
  
  // Likelihood
  for (i in 1:N) {
    y[i] ~ normal(mu + beta_b * x1[i] + beta_w * x2[i] + sp_eff[group[i]] + phy_eff[group[i]], sigma);
  }
}

  }
